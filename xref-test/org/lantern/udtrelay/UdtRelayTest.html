120     private void startProxyServer(final int port,
121         final HandshakeHandlerFactory ksm, final boolean ssl) throws Exception {
122         
123         // We configure the proxy server to always return a cache hit with 
124         // the same generic response.
125         final Thread t = new Thread(new Runnable() {
126 
127             @Override
128             public void run() {
129                 if (ssl) {
130                     org.jboss.netty.util.Timer timer = 
131                         new org.jboss.netty.util.HashedWheelTimer();
132                     final org.lantern.HttpProxyServer server = 
133                         new StatsTrackingDefaultHttpProxyServer(
134                         new HttpResponseFilters() {
135                             @Override
136                             public HttpFilter getFilter(String arg0) {
137                                 return null;
138                             }
139                         }, null, null,
140                         new NioClientSocketChannelFactory(), timer,
141                         new NioServerSocketChannelFactory(), ksm, null,
142                         null) {
143                             @Override
144                             public int getPort() {
145                                 return port;
146                             }
147                     };
148                     try {
149                         server.start();
150                     } catch (final Exception e) {
151                         log.error("Error starting server!!", e);
152                     }
153                 } else {
154                     final org.littleshoot.proxy.HttpProxyServer server =
155                             new DefaultHttpProxyServer(port);
156                     try {
157                         server.start();
158                     } catch (final Exception e) {
159                         log.error("Error starting server!!", e);
160                     }
161                 }
162             }
163         }, "Relay-to-Proxy-Test-Thread");
164         t.setDaemon(true);
165         t.start();
166         if (!LanternUtils.waitForServer(port, 6000)) {
167             fail("Could not start local test proxy server!!");
168         }

171     private String hitProxyDirect(final int proxyPort) throws Exception {
172         final Socket sock = new Socket();
173         sock.connect(new InetSocketAddress("127.0.0.1", proxyPort));
174         
175         sock.getOutputStream().write(REQUEST.getBytes());
176         
177         final InputStream is = sock.getInputStream();
178         sock.setSoTimeout(4000);
179         final BufferedReader br = new BufferedReader(new InputStreamReader(is));
180         final StringBuilder sb = new StringBuilder();
181         String cur = br.readLine();
182         sb.append(cur);
183         while(StringUtils.isNotBlank(cur)) {
184             System.err.println(cur);
185             cur = br.readLine();
186             if (!cur.startsWith("x-amz-") && !cur.startsWith("Date")) {
187                 sb.append(cur);
188                 sb.append("\n");
189             }
190         }
191         final String response = sb.toString();
192         sock.close();
193         assertTrue("Unexpected response "+response, 
194             response.startsWith("HTTP/1.1 200 OK"));
195 
196         return response;

254     private void hitRelayUdtNetty(
255         final DummyChannel browserToProxyChannel, final HttpRequest request,
256         final FiveTuple ft, final LanternTrustStore trustStore) throws Exception {
257         browserToProxyChannel.setReadable(false);
258         
259         final Bootstrap boot = new Bootstrap();
260         final ThreadFactory connectFactory = 
261                 Threads.newNonDaemonThreadFactory("connect");
262         final NioEventLoopGroup connectGroup = new NioEventLoopGroup(1,
263                 connectFactory, NioUdtProvider.BYTE_PROVIDER);
264 
265         try {
266             boot.group(connectGroup)
267                 .channelFactory(NioUdtProvider.BYTE_CONNECTOR)
268                 .option(ChannelOption.SO_REUSEADDR, true)
269                 .handler(new ChannelInitializer<UdtChannel>() {
270                     @Override
271                     public void initChannel(final UdtChannel ch)
272                             throws Exception {
273                         final ChannelPipeline p = ch.pipeline();
274                         final SSLEngine engine = 
275                             trustStore.getSslContext().createSSLEngine();
276                         
277                         //SSLEngine serverEngine = sslc.createSSLEngine();
278                         engine.setUseClientMode(true);
279 
280                         p.addLast("ssl", new SslHandler(engine));
281                         p.addLast(
282                             //new LoggingHandler(LogLevel.INFO),
283                             new HttpResponseClientHandler(
284                                 browserToProxyChannel, request));
285                     }
286                 });
287             /*
288             try {
289                 boot.bind(ft.getLocal()).sync();
290             } catch (final InterruptedException e) {
291                 log.error("Could not sync on bind? Reuse address no working?", e);
292             }
293             */
294             
295             // Start the client.
296             final ChannelFuture f = 
297                 boot.connect(ft.getRemote(), ft.getLocal()).sync();
298             
299             log.debug("Got connected!! Encoding request");
300             f.channel().write(encoder.encode(request)).sync();
301             
302             synchronized(browserToProxyChannel) {
303                 if (browserToProxyChannel.message.length() == 0) {
304                     browserToProxyChannel.wait(4000);
305                 }
306             }
307             
308             assertTrue("Unexpected response. Beginning is:\n"+
309                     browserToProxyChannel.message.substring(0, 200),
310                     // Can apparently get HTTP 1.0 responses in some cases...
311                     browserToProxyChannel.message.startsWith("HTTP/1.1 200 OK") ||
312                     browserToProxyChannel.message.startsWith("HTTP/1.0 200 OK"));
313             
314             f.channel().close();
315             
316         } finally {
317             // Shut down the event loop to terminate all threads.
318             //boot.shutdown();
319         }
320     }
