158     private String trySite(final HttpClient client, final String uri)
159         throws Exception {
160         final HttpGet get = new HttpGet();
161         final String fullUri = "https://"+uri;
162         log.info("Hitting URI: {}", fullUri);
163         get.setURI(new URI(fullUri));
164 
165         final HttpResponse response = client.execute(get);
166         final int code = response.getStatusLine().getStatusCode();
167         final HttpEntity entity = response.getEntity();
168         final String body =
169             IOUtils.toString(entity.getContent()).toLowerCase();
170         EntityUtils.consume(entity);
171 
172         if (code < 200 || code > 299) {
173             // We use this method both for calls that should succeed and
174             // calls that should fail, so this is expected.
175             log.debug("Non-200 response code: "+code+" for "+uri+
176                 " with body:\n"+body);
177         }
178         get.reset();
179         return body;
180     }
