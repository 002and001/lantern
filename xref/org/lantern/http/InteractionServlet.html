158     protected void processRequest(final HttpServletRequest req,
159         final HttpServletResponse resp) {
160         LanternUtils.addCSPHeader(resp);
161         final String uri = req.getRequestURI();
162         log.debug("Received URI: {}", uri);
163         final String interactionStr = StringUtils.substringAfterLast(uri, "/");
164         if (StringUtils.isBlank(interactionStr)) {
165             log.debug("blank interaction");
166             HttpUtils.sendClientError(resp, "blank interaction");
167             return;
168         }
169 
170         log.debug("Headers: "+HttpUtils.getRequestHeaders(req));
171 
172         if (!"XMLHttpRequest".equals(req.getHeader("X-Requested-With"))) {
173             log.debug("invalid X-Requested-With");
174             HttpUtils.sendClientError(resp, "invalid X-Requested-With");
175             return;
176         }
177 
178         if (!SecurityUtils.constantTimeEquals(model.getXsrfToken(),
179                 req.getHeader("X-XSRF-TOKEN"))) {
180             log.debug("X-XSRF-TOKEN wrong: got {} expected {}", req.getHeader("X-XSRF-TOKEN"), model.getXsrfToken());
181             HttpUtils.sendClientError(resp, "invalid X-XSRF-TOKEN");
182             return;
183         }
184 
185         final int cl = req.getContentLength();
186         String json = "";
187         if (cl > 0) {
188             try {
189                 json = IOUtils.toString(req.getInputStream());
190             } catch (final IOException e) {
191                 log.error("Could not parse json?");
192             }
193         }
194 
195         log.debug("Body: '"+json+"'");
196 
197         final Interaction inter =
198             Interaction.valueOf(interactionStr.toUpperCase());
199 
200         if (inter == Interaction.CLOSE) {
201             if (handleClose(json)) {
202                 return;
203             }
204         }
205 
206         if (inter == Interaction.URL) {
207             final String url = JsonUtils.getValueFromJson("url", json);
208             if (!StringUtils.startsWith(url, "http://") &&
209                 !StringUtils.startsWith(url, "https://")) {
210                 log.error("http(s) url expected, got {}", url);
211                 HttpUtils.sendClientError(resp, "http(s) urls only");
212                 return;
213             }
214             try {
215                 new URL(url);
216             } catch (MalformedURLException e) {
217                 log.error("invalid url: {}", url);
218                 HttpUtils.sendClientError(resp, "invalid url");
219                 return;
220             }
221 
222             final String cmd;
223             if (SystemUtils.IS_OS_MAC_OSX) {
224                 cmd = "open";
225             } else if (SystemUtils.IS_OS_LINUX) {
226                 cmd = "gnome-open";
227             } else if (SystemUtils.IS_OS_WINDOWS) {
228                 cmd = "start";
229             } else {
230                 log.error("unsupported OS");
231                 HttpUtils.sendClientError(resp, "unsupported OS");
232                 return;
233             }
234             try {
235                 if (SystemUtils.IS_OS_WINDOWS) {
236                     // On Windows, we have to quote the url to allow for
237                     // e.g. ? and & characters in query string params.
238                     // To quote the url, we supply a dummy first argument,
239                     // since otherwise start treats the first argument as a
240                     // title for the new console window when it's quoted.
241                     LanternUtils.runCommand(cmd, "\"\"", "\""+url+"\"");
242                 } else {
243                     // on OS X and Linux, special characters in the url make
244                     // it through this call without our having to quote them.
245                     LanternUtils.runCommand(cmd, url);
246                 }
247             } catch (IOException e) {
248                 log.error("open url failed");
249                 HttpUtils.sendClientError(resp, "open url failed");
250                 return;
251             }
252             return;
253         }
254 
255         final Modal modal = this.model.getModal();
256 
257         log.debug("processRequest: modal = {}, inter = {}, mode = {}", 
258             modal, inter, this.model.getSettings().getMode());
259         
260         if (handleExceptionalInteractions(modal, inter, json)) {
261             return; 
262         }
263 
264         Modal switchTo = null;
265         try {
266             // XXX a map would make this more robust
267             switchTo = Modal.valueOf(interactionStr);
268         } catch (IllegalArgumentException e) { }
269         if (switchTo != null && switchModals.contains(switchTo)) {
270             if (!switchTo.equals(modal)) {
271                 if (!switchModals.contains(modal)) {
272                     this.internalState.setLastModal(modal);
273                 }
274                 Events.syncModal(model, switchTo);
275             }
276             return;
277         }
278 
279         switch (modal) {
280         case welcome:
281             this.model.getSettings().setMode(Mode.unknown);
282             switch (inter) {
283             case GET:
284                 log.debug("Setting get mode");
285                 handleSetModeWelcome(Mode.get);
286                 break;
287             case GIVE:
288                 log.debug("Setting give mode");
289                 handleSetModeWelcome(Mode.give);
290                 break;
291             }
292             break;
293         case authorize:
294            log.debug("Processing authorize modal...");
295             this.internalState.setModalCompleted(Modal.authorize);
296             this.internalState.advanceModal(null);
297             break;
298         case finished:
299             this.internalState.setCompletedTo(Modal.finished);
300             switch (inter) {
301             case CONTINUE:
302                 log.debug("Processing continue");
303                 this.model.setShowVis(true);
304                 Events.sync(SyncPath.SHOWVIS, true);
305                 this.internalState.setModalCompleted(Modal.finished);
306                 this.internalState.advanceModal(null);
307                 break;
308             case SET:
309                 log.debug("Processing set in finished modal...applying JSON\n{}", 
310                         json);
311                 applyJson(json);
312                 break;
313             default:
314                 log.error("Did not handle interaction {} for modal {}", inter, modal);
315                 HttpUtils.sendClientError(resp, 
316                         "Interaction not handled for modal: "+modal+
317                         " and interaction: "+inter);
318                 break;
319             }
320             break;
321         case firstInviteReceived:
322             log.error("Processing invite received...");
323             break;
324         case lanternFriends:
325             this.internalState.setCompletedTo(Modal.lanternFriends);
326             switch (inter) {
327             case FRIEND:
328                 addFriend(json);
329                 break;
330             case REJECT:
331                 removeFriend(json);
332                 break;
333             case CONTINUE:
334                 // This dialog always passes continue as of this writing and
335                 // not close.
336             case CLOSE:
337                 log.debug("Processing continue/close for friends dialog");
338                 if (this.model.isSetupComplete()) {
339                     Events.syncModal(model, Modal.none);
340                 } else {
341                     this.internalState.setModalCompleted(Modal.lanternFriends);
342                     this.internalState.advanceModal(null);
343                 }
344                 break;
345             default:
346                 log.error("Did not handle interaction {} for modal {}", inter, modal);
347                 HttpUtils.sendClientError(resp,
348                     "Interaction not handled for modal: "+modal+
349                     " and interaction: "+inter);
350                 break;
351             }
352             break;
353         case none:
354             break;
355         case notInvited:
356             switch (inter) {
357             case RETRY:
358                 Events.syncModal(model, Modal.authorize);
359                 break;
360             case REQUESTINVITE:
361                 Events.syncModal(model, Modal.requestInvite);
362                 break;
363             default:
364                 log.error("Unexpected interaction: " + inter);
365                 break;
366             }
367             break;
368         case proxiedSites:
369             this.internalState.setCompletedTo(Modal.proxiedSites);
370             switch (inter) {
371             case CONTINUE:
372                 if (this.model.isSetupComplete()) {
373                     Events.syncModal(model, Modal.none);
374                 } else {
375                     this.internalState.setModalCompleted(Modal.proxiedSites);
376                     this.internalState.advanceModal(null);
377                 }
378                 break;
379             case LANTERNFRIENDS:
380                 log.debug("Processing lanternFriends from proxiedSites");
381                 Events.syncModal(model, Modal.lanternFriends);
382                 break;
383             case SET:
384                 if (!model.getSettings().isSystemProxy()) {
385                     String msg = "Because you are using manual proxy "
386                             + "configuration, you may have to restart your "
387                             + "browser for your updated proxied sites list "
388                             + "to take effect.";
389                     model.addNotification(msg, MessageType.info, 30);
390                     Events.sync(SyncPath.NOTIFICATIONS,
391                             model.getNotifications());
392                 }
393                 applyJson(json);
394                 break;
395             case SETTINGS:
396                 log.debug("Processing settings from proxiedSites");
397                 Events.syncModal(model, Modal.settings);
398                 break;
399             default:
400                 log.error("Did not handle interaction {} for modal {}", inter, modal);
401                 HttpUtils.sendClientError(resp, "unexpected interaction for proxied sites");
402                 break;
403             }
404             break;
405         case requestInvite:
406             log.info("Processing request invite");
407             switch (inter) {
408             case CANCEL:
409                 this.internalState.setModalCompleted(Modal.requestInvite);
410                 this.internalState.advanceModal(Modal.notInvited);
411                 break;
412             case CONTINUE:
413                 applyJson(json);
414                 this.internalState.setModalCompleted(Modal.proxiedSites);
415                 //TODO: need to do something here
416                 this.internalState.advanceModal(null);
417                 break;
418             default:
419                 log.error("Did not handle interaction {} for modal {}", inter, modal);
420                 HttpUtils.sendClientError(resp, "unexpected interaction for request invite");
421                 break;
422             }
423             break;
424         case requestSent:
425             log.debug("Process request sent");
426             break;
427         case settings:
428             switch (inter) {
429             case GET:
430                 log.debug("Setting get mode");
431                 // Only deal with a mode change if the mode has changed!
432                 if (modelService.getMode() == Mode.give) {
433                     // Break this out because it's set in the subsequent 
434                     // setMode call
435                     final boolean everGet = model.isEverGetMode();
436                     this.modelService.setMode(Mode.get);
437                     if (!everGet) {
438                         // need to do more setup to switch to get mode from 
439                         // give mode
440                         model.setSetupComplete(false);
441                         model.setModal(Modal.proxiedSites);
442                         Events.syncModel(model);
443                     } else {
444                         // This primarily just triggers a setup complete event,
445                         // which triggers connecting to proxies, setting up
446                         // the local system proxy, etc.
447                         model.setSetupComplete(true);
448                     }
449                 }
450                 break;
451             case GIVE:
452                 log.debug("Setting give mode");
453                 this.modelService.setMode(Mode.give);
454                 break;
455             case CLOSE:
456                 log.debug("Processing settings close");
457                 Events.syncModal(model, Modal.none);
458                 break;
459             case SET:
460                 log.debug("Processing set in setting...applying JSON\n{}", json);
461                 applyJson(json);
462                 break;
463             case RESET:
464                 log.debug("Processing reset");
465                 Events.syncModal(model, Modal.confirmReset);
466                 break;
467             case PROXIEDSITES:
468                 log.debug("Processing proxied sites in settings");
469                 Events.syncModal(model, Modal.proxiedSites);
470                 break;
471             case LANTERNFRIENDS:
472                 log.debug("Processing friends in settings");
473                 Events.syncModal(model, Modal.lanternFriends);
474                 break;
475 
476             default:
477                 log.error("Did not handle interaction {} for modal {}", inter, modal);
478                 HttpUtils.sendClientError(resp, 
479                         "Interaction not handled for modal: "+modal+
480                         " and interaction: "+inter);
481                 break;
482             }
483             break;
484         case settingsLoadFailure:
485             switch (inter) {
486             case RETRY:
487                 modelIo.reload();
488                 Events.sync(SyncPath.NOTIFICATIONS, model.getNotifications());
489                 Events.syncModal(model, model.getModal());
490                 break;
491             case RESET:
492                 backupSettings();
493                 Events.syncModal(model, Modal.welcome);
494                 break;
495             default:
496                 log.error("Did not handle interaction {} for modal {}", inter, modal);
497                 break;
498             }
499             break;
500         case systemProxy:
501             this.internalState.setCompletedTo(Modal.systemProxy);
502             switch (inter) {
503             case CONTINUE:
504                 log.debug("Processing continue in systemProxy", json);
505                 applyJson(json);
506                 Events.sync(SyncPath.SYSTEMPROXY, model.getSettings().isSystemProxy());
507                 this.internalState.setModalCompleted(Modal.systemProxy);
508                 this.internalState.advanceModal(null);
509                 break;
510             default:
511                 log.error("Did not handle interaction {} for modal {}", inter, modal);
512                 HttpUtils.sendClientError(resp, "error setting system proxy pref");
513                 break;
514             }
515             break;
516         case updateAvailable:
517 
518             switch (inter) {
519             case CLOSE:
520                 this.internalState.setModalCompleted(Modal.updateAvailable);
521                 this.internalState.advanceModal(null);
522                 break;
523             default:
524                 log.error("Did not handle interaction {} for modal {}", inter, modal);
525                 break;
526             }
527             break;
528         case authorizeLater:
529             log.error("Did not handle interaction {} for modal {}", inter, modal);
530             break;
531         case confirmReset:
532             log.debug("Handling confirm reset interaction");
533             switch (inter) {
534             case CANCEL:
535                 log.debug("Processing cancel");
536                 Events.syncModal(model, Modal.settings);
537                 break;
538             case RESET:
539                 handleReset();
540                 Events.syncModel(this.model);
541                 break;
542             default:
543                 log.error("Did not handle interaction {} for modal {}", inter, modal);
544                 HttpUtils.sendClientError(resp, 
545                         "Interaction not handled for modal: "+modal+
546                         " and interaction: "+inter);
547             }
548             break;
549         case about: // fall through on purpose
550         case sponsor:
551             switch (inter) {
552             case CLOSE:
553                 Events.syncModal(model, this.internalState.getLastModal());
554                 break;
555             default:
556                 HttpUtils.sendClientError(resp, "invalid interaction "+inter);
557             }
558             break;
559         case contact:
560             switch(inter) {
561             case CONTINUE:
562                 String msg;
563                 MessageType messageType;
564                 try {
565                     lanternFeedback.submit(json,
566                         this.model.getProfile().getEmail());
567                     msg = "Thank you for contacting Lantern.";
568                     messageType = MessageType.info;
569                 } catch(Exception e) {
570                     log.error("Error submitting contact form: {}", e);
571                     msg = "Error sending message. Please check your "+
572                         "connection and try again.";
573                     messageType = MessageType.error;
574                 }
575                 model.addNotification(msg, messageType, 30);
576                 Events.sync(SyncPath.NOTIFICATIONS, model.getNotifications());
577             // fall through because this should be done in both cases:
578             case CANCEL:
579                 Events.syncModal(model, this.internalState.getLastModal());
580                 break;
581             default:
582                 HttpUtils.sendClientError(resp, "invalid interaction "+inter);
583 
584             }
585             break;
586         case giveModeForbidden:
587             if (inter == Interaction.CONTINUE) {
588                 //  need to do more setup to switch to get mode from give mode
589                 model.getSettings().setMode(Mode.get);
590                 model.setSetupComplete(false);
591                 this.internalState.advanceModal(null);
592                 Events.syncModal(model, Modal.proxiedSites);
593                 Events.sync(SyncPath.SETUPCOMPLETE, false);
594             }
595             break;
596         default:
597             log.error("No matching modal for {}", modal);
598         }
599         this.modelIo.write();
600     }

602     private void setFriendStatus(String json, Status status) {
603         final String email = JsonUtils.getValueFromJson("email", json).toLowerCase();
604         Friends friends = model.getFriends();
605         Friend friend = friends.get(email);
606         if (friend != null && friend.getStatus() == Status.friend) {
607           model.addNotification("You have already friended "+email+".",
608               MessageType.info, 30);
609           Events.sync(SyncPath.NOTIFICATIONS, model.getNotifications());
610           return;
611         }
612         if (friend == null || friend.getStatus() == Status.rejected) {
613             friend = modelUtils.makeFriend(email);
614             if (status == Status.friend) {
615                 model.addNotification("An email will be sent to "+email+" "+
616                     "with a notification that you friended them. "+
617                     "If they do not yet have a Lantern invite, they will "+
618                     "be invited when the network can accommodate them.",
619                     MessageType.info, 30);
620                 Events.sync(SyncPath.NOTIFICATIONS, model.getNotifications());
621                 inviteQueue.invite(friend);
622             }
623         }
624         friend.setStatus(status);
625         friends.setNeedsSync(true);
626         Events.asyncEventBus().post(new FriendStatusChangedEvent(friend));
627         Events.sync(SyncPath.FRIENDS, friends.getFriends());
628     }

634     private void addFriend(String json) {
635         setFriendStatus(json, Status.friend);
636         final String email = JsonUtils.getValueFromJson("email", json).toLowerCase();
637 
638         try {
639             //if they have requested a subscription to us, we'll accept it.
640             this.xmppHandler.subscribed(email);
641 
642             // We also automatically subscribe to them in turn so we know about
643             // their presence.
644             this.xmppHandler.subscribe(email);
645         } catch (IllegalStateException e) {
646             log.info("IllegalStateException while friending (you are probably offline)", e);
647             return;
648         }
649     }

660     private boolean handleExceptionalInteractions(
661             final Modal modal, final Interaction inter, final String json) {
662         boolean handled = false;
663         Map<String, Object> map;
664         Boolean notify;
665         switch(inter) {
666             case EXCEPTION:
667                 handleException(json);
668                 handled = true;
669                 break;
670             case UNEXPECTEDSTATERESET:
671                 log.debug("Handling unexpected state reset.");
672                 backupSettings();
673                 handleReset();
674                 Events.syncModel(this.model);
675             // fall through because this should be done in both cases:
676             case UNEXPECTEDSTATEREFRESH:
677                 try {
678                     map = jsonToMap(json);
679                 } catch(Exception e) {
680                     log.error("Bad json payload in inter '{}': {}", inter, json);
681                     return true;
682                 }
683                 notify = (Boolean)map.get("notify");
684                 if(notify) {
685                     try {
686                         lanternFeedback.submit((String)map.get("report"),
687                             this.model.getProfile().getEmail());
688                     } catch(Exception e) {
689                         log.error("Could not submit unexpected state report: {}\n {}",
690                             e.getMessage(), (String)map.get("report"));
691                     }
692                 }
693                 handled = true;
694                 break;
695         }
696         return handled;
697     }

699     private void handleException(final String json) {
700         StringBuilder logMessage = new StringBuilder();
701         Map<String, Object> map;
702         try {
703             map = jsonToMap(json);
704         } catch(Exception e) {
705             log.error("UI Exception (unable to parse json)");
706             return;
707         }
708         for(Map.Entry<String, Object> entry : map.entrySet()) {
709             logMessage.append(
710                 String.format("\t%s: %s\n", 
711                     entry.getKey(), entry.getValue()
712                 )
713             );
714         }
715         log.error("UI Exception:\n {}", logMessage.toString());
716     }

718     private Map<String, Object> jsonToMap(final String json) 
719             throws JsonParseException, JsonMappingException, IOException {
720         final ObjectMapper om = new ObjectMapper();
721         Map<String, Object> map;
722         map = om.readValue(json, Map.class);
723         return map;
724     }

727     private boolean handleClose(String json) {
728         if (StringUtils.isBlank(json)) {
729             return false;
730         }

821 
822         if (set.getMode() == null || set.getMode() == Mode.unknown) {
823             if (censored.isCensored()) {

825             } else {
826                 set.setMode(Mode.give);
