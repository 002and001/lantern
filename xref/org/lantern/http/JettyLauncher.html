79      public void start(final int port) {
80          final QueuedThreadPool qtp = new QueuedThreadPool();
81          qtp.setMinThreads(5);
82          qtp.setMaxThreads(200);
83          qtp.setName("Lantern-Jetty-Thread");
84          qtp.setDaemon(true);
85          server.setThreadPool(qtp);
86          final String apiName = "Lantern-API";
87          final ContextHandlerCollection contexts = 
88              new ContextHandlerCollection();
89  
90          String prefix = StaticSettings.getPrefix();
91  
92          final ServletContextHandler contextHandler = newContext(prefix, apiName);
93  
94          contexts.addHandler(contextHandler);
95  
96          final String resourceBase;
97          final String app = "./lantern-ui/app";
98          final File appFile = new File(app);
99          if (appFile.isDirectory()) {
100             resourceBase = app;
101         } else {
102             resourceBase = "./lantern-ui";
103         }
104         contextHandler.setResourceBase(resourceBase);
105         
106         server.setHandler(contexts);
107         server.setStopAtShutdown(true);
108         
109         final SelectChannelConnector connector = 
110             new SelectChannelConnector();
111         connector.setPort(port);
112         connector.setMaxIdleTime(60 * 1000);
113         connector.setLowResourcesMaxIdleTime(30 * 1000);
114         connector.setLowResourcesConnections(2000);
115         connector.setAcceptQueueSize(5000);
116 
117 
118         if (this.model.getSettings().isBindToLocalhost()) {
119             log.info("Binding to localhost");
120             connector.setHost("127.0.0.1");
121         } else {
122             log.info("Binding to any address");
123         }
124         connector.setName(apiName);
125         
126         this.server.setConnectors(new Connector[]{connector});
127 
128         final CometdServlet cometdServlet = new LanternCometdServlet();
129 
130         final ServletHolder cometd = new ServletHolder(cometdServlet);
131         cometd.setInitParameter("jsonContext", 
132             "org.lantern.SettingsJSONContextServer");
133 
134         // An init order of -1 means to initialize upon first use.
135         cometd.setInitOrder(-11);
136         contextHandler.addServlet(cometd, "/cometd/*");
137         
138         final ServletHolder ds = new ServletHolder(new DefaultServlet() {
139 
140             private static final long serialVersionUID = 4335271390548389540L;
141 
142             @Override
143             protected void doGet(final HttpServletRequest req, 
144                 final HttpServletResponse resp) throws ServletException, 
145                 IOException {
146                 final String uri = req.getPathInfo();
147                 log.debug("Processing get request for static file: "+uri);
148                 if (uri.startsWith("/proxy_on")) {
149                     writeFileToResponse(resp, Proxifier.PROXY_ON);
150                 } else if (uri.startsWith("/proxy_off")) {
151                     writeFileToResponse(resp, Proxifier.PROXY_OFF);
152                 } else if (uri.startsWith("/proxy_all")) {
153                     writeFileToResponse(resp, Proxifier.PROXY_ALL);
154                 } else if (uri.startsWith("/proxy_google")) {
155                     writeFileToResponse(resp, Proxifier.PROXY_GOOGLE);
156                 } else {
157                     resp.addCookie(new Cookie("XSRF-TOKEN", model.getXsrfToken()));
158                     LanternUtils.addCSPHeader(resp);
159                     super.doGet(req, resp);
160                 }
161             }
162         });
163         ds.setInitParameter("cacheControl", "no-cache");
164         ds.setInitParameter("aliases", "true");
165 
166         ds.setInitOrder(1);
167         contextHandler.addServlet(ds, "/*");
168         
169         if (this.redirectServlet != null) {
170             final ServletHolder settings = new ServletHolder(redirectServlet);
171             settings.setInitOrder(-1);
172             contextHandler.addServlet(settings, "/oauth/");
173         }
174         
175         if (this.interactionServlet != null) {
176             final ServletHolder interactionServletHolder = 
177                 new ServletHolder(this.interactionServlet);
178             interactionServletHolder.setInitOrder(-1);
179             contextHandler.addServlet(interactionServletHolder, apiPath());
180         }
181         
182         if (this.photoServlet != null) {
183             final ServletHolder photo = new ServletHolder(this.photoServlet);
184             photo.setInitOrder(-1);
185             contextHandler.addServlet(photo, "/photo/*");
186         }
187 
188         
189         final BayeuxInitializer bi = new BayeuxInitializer(this.syncer);
190         final ServletHolder bayeux = new ServletHolder(bi);
191         bayeux.setInitParameter("jsonContext", 
192             "org.cometd.server.JacksonJSONContextServer");
193         bayeux.setInitOrder(-1);
194         contextHandler.getServletHandler().addServlet(bayeux);
195         
196         if (!this.model.getSettings().isBindToLocalhost()) {
197             final CrossOriginFilter filter = new CrossOriginFilter();
198             final FilterHolder filterHolder = new FilterHolder(filter);
199             //filterHolder.setInitParameter("allowedOrigins", "http://fiddle.jshell.net/");
200             filterHolder.setInitParameter("allowedOrigins", "*");
201         }
202 
203         final Thread serve = new Thread(new Runnable() {
204 
205             @Override
206             public void run() {
207                 try {
208                     server.start();
209                     server.join();
210                 } catch (final Exception e) {
211                     log.error("Exception on HTTP server");
212                 }
213             }
214             
215         }, "HTTP-Server-Thread");
216         serve.setDaemon(true);
217         serve.start();
218     }

225     private void writeFileToResponse(final HttpServletResponse resp,
226         final File file) {
227         InputStream is = null;
228         OutputStream os = null;
229         try {
230             is = new FileInputStream(file);
231             os = resp.getOutputStream();
232             resp.setContentLength((int) file.length());
233             resp.setContentType("application/x-ns-proxy-autoconfig");
234             IOUtils.copyLarge(is, os);
235         } catch (final IOException e) {
236             log.error("Could not write response?", e);
237         } finally {
238             IOUtils.closeQuietly(is);
239             IOUtils.closeQuietly(os);
240         }
241     }

254     private ServletContextHandler newContext(final String path,
255         final String name) {
256         final ServletContextHandler context = 
257             new ServletContextHandler(ServletContextHandler.SESSIONS);
258         
259         final Map<String, String> params = context.getInitParams();
260         params.put("org.eclipse.jetty.servlet.Default.gzip", "false");
261         params.put("org.eclipse.jetty.servlet.Default.welcomeServlets", "false");
262         params.put("org.eclipse.jetty.servlet.Default.dirAllowed", "false");
263         params.put("org.eclipse.jetty.servlet.Default.aliases", "false");
264         params.put("jsonContext", "org.cometd.server.JacksonJSONContextServer");
265         context.setContextPath(path);
266         context.setConnectorNames(new String[] {name});
267         return context;
268     }
