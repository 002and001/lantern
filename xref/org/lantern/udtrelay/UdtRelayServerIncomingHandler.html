51      @Override
52      public void inboundBufferUpdated(final ChannelHandlerContext ctx,
53              final ByteBuf in) throws IOException {
54          // Just write incoming HTTP request bytes to the outgoing connection 
55          // to the destination server.
56          log.debug("Inbound buffer with bytes: {}", in.readableBytes());
57          //final ByteBuf out = outboundChannel.outboundByteBuffer();
58          //out.writeBytes(in);
59          if (this.sock.isConnected()) {
60              final OutputStream os = sock.getOutputStream();
61              ByteBufInputStream is = null;
62              try  {
63                  is = new ByteBufInputStream(in);
64                  final byte[] incoming = new byte[is.available()];
65                  
66                  final int read = is.read(incoming);
67                  if (read != incoming.length) {
68                      log.warn("Didn't read all the available bytes?!?");
69                  }
70                  // Note this will typically be encrypted data here, but is the
71                  // HTTP request.
72                  os.write(incoming);
73                  os.flush();
74                  ctx.channel().read();
75              } finally {
76                  IOUtils.closeQuietly(is);
77              }
78  
79          } else {
80              log.error("Socket not connected?");
81              log.debug("Failed to flush data?");
82              IOUtils.closeQuietly(sock);
83              NettyUtils.closeOnFlush(ctx.channel());
84          }
85      }

91                  @Override

95                          log.debug("Flushed data on outbound connection!!");
96                          // Flushed out data - start to read the next chunk

127     private static final int LARGE_BUFFER_SIZE = 1024 * 16;

129     private void socketRelay(final Channel inboundChannel) throws IOException {
130         this.sock = new Socket();
131         try {
132             sock.connect(new InetSocketAddress(
133                 LanternClientConstants.LOCALHOST, localProxyPort), 30*1000);
134             inboundChannel.read();
135         } catch (final IOException e) {
136             log.warn("Outbound channel connection failed!");
137             // Close the connection if the connection attempt has 
138             // failed.
139             NettyUtils.closeOnFlush(inboundChannel);
140         }
141         
142         readFromSocketThread(inboundChannel);
143     }
144     
145     private void readFromSocketThread(final Channel inboundChannel) 
146         throws IOException {
147         final InputStream is = this.sock.getInputStream();
148         
149         final Runnable runner = new Runnable() {
150             public void run() {
151                 try {
152                     copyLarge(is, inboundChannel, LARGE_BUFFER_SIZE);
153                 } catch (final IOException e) {
154                     // This will happen if the other side just closes the
155                     // socket, for example.
156                     log.debug("Error copying socket data on", e);
157                 } catch (final Throwable t) {
158                     log.warn("Error copying socket data on", t);
159                 } finally {
160                     // Flush to be sure we've written everything.
161                     NettyUtils.closeOnFlush(inboundChannel);
162                     
163                     // This happens on JVM shutdown, for example.
164                     log.info("Closing socket...already closed streams...");
165                     IOUtils.closeQuietly(sock);
166                 }
167             }
168         };
169         final Thread thread = new Thread(runner,
170                 "RelayingSocketHandler-Thread-"
171                         + runner.hashCode());
172         thread.setDaemon(true);
173         thread.start();
174     }

176     private void copyLarge(final InputStream input, final Channel inboundChannel,
177             final int bufferSize) throws IOException {
178         final byte[] buffer = new byte[bufferSize];
179         int n = 0;
180         while (-1 != (n = input.read(buffer))) {
181             final ByteBuf wrapped = Unpooled.wrappedBuffer(buffer, 0, n);
182             //log.debug("Writing to inbound channel: {}", 
183             //    wrapped.toString(LanternConstants.UTF8));
184             inboundChannel.write(wrapped);
185             
186             // We need to flush and sync here, as otherwise the buffer will
187             // get overwritten with new data.
188             try {
189                 inboundChannel.flush().sync();
190             } catch (InterruptedException e) {
191                 log.error("Error flushing", e);
192                 throw new RuntimeException("Error flushing", e);
193             }
194         }
195         log.debug("Copied bytes...");
196     }

198     private void netty4ProxyConnect(final Channel inboundChannel) {
199 
200         // Start the connection attempt.
201         final Bootstrap clientBootstrapFromRelayToBackendServer = 
202              new Bootstrap();
203         clientBootstrapFromRelayToBackendServer.group(inboundChannel.eventLoop())
204             .channel(NioSocketChannel.class)
205             .handler(new UdtRelayServerBackendHandler(inboundChannel))
206             .option(ChannelOption.AUTO_READ, false);
207         
208         final ChannelFuture cf = 
209             clientBootstrapFromRelayToBackendServer.connect(
210                 LanternClientConstants.LOCALHOST, localProxyPort);
211         outboundChannel = cf.channel();
212         
213         cf.addListener(new ChannelFutureListener() {
214             @Override
215             public void operationComplete(ChannelFuture future) 
216                 throws Exception {
217                 if (future.isSuccess()) {
218                     log.debug("Outbound channel connected!");
219                     // Connection complete start to read first data
220                     inboundChannel.read();
221                     log.debug("Reading from inbound channel");
222                 } else {
223                     log.warn("Outbound channel connection failed!");
224                     // Close the connection if the connection attempt has 
225                     // failed.
226                     inboundChannel.close();
227                 }
228             }
229         });
230     }
