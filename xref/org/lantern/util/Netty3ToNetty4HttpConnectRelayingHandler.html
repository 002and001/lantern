16  /**
17   * Class that simply relays traffic the channel this is connected to to 
18   * another channel passed in to the constructor.
19   */
20  @Sharable
21  public class Netty3ToNetty4HttpConnectRelayingHandler 
22      extends SimpleChannelUpstreamHandler {
23      
24      private static final Logger LOG = 
25          LoggerFactory.getLogger(Netty3ToNetty4HttpConnectRelayingHandler.class);
26      
27      /**
28       * The channel to relay to. This could be a connection from the browser
29       * to the proxy or it could be a connection from the proxy to an external
30       * site.
31       */
32      private final io.netty.channel.Channel netty4RelayChannel;
33  
34      private final ChannelGroup channelGroup;
35  
36      /**
37       * Creates a new {@link Netty3ToNetty4HttpConnectRelayingHandler} with the 
38       * specified connection to relay to.
39       * 
40       * @param netty4Channel The channel to relay messages to.
41       * @param channelGroup The group of channels to close on shutdown.
42       */
43      public Netty3ToNetty4HttpConnectRelayingHandler(
44          final io.netty.channel.Channel netty4Channel, 
45          final ChannelGroup channelGroup) {
46          // Fail fast if these are null.
47          if (netty4Channel == null) {
48              throw new NullPointerException("Relay channel is null!");
49          }
50          this.netty4RelayChannel = netty4Channel;
51          this.channelGroup = channelGroup;
52      }
53  
54      @Override
55      public void messageReceived(final ChannelHandlerContext ctx, 
56          final MessageEvent e) throws Exception {
57          final ChannelBuffer msg = (ChannelBuffer) e.getMessage();
58          if (netty4RelayChannel.isOpen()) {
59              final io.netty.channel.ChannelFutureListener logListener = 
60                  new io.netty.channel.ChannelFutureListener() {
61                      @Override
62                      public void operationComplete(
63                          final io.netty.channel.ChannelFuture cf)
64                          throws Exception {
65                          LOG.debug("Finished writing data on CONNECT channel");
66                      }
67                  };
68              netty4RelayChannel.write(
69                  NettyUtils.channelBufferToByteBuf(msg)).addListener(logListener);
70          }
71          else {
72              LOG.debug("Channel not open. Connected? {}", netty4RelayChannel.isOpen());
73              // This will undoubtedly happen anyway, but just in case.
74              ProxyUtils.closeOnFlush(e.getChannel());
75          }
76      }
77      
78      @Override
79      public void channelOpen(final ChannelHandlerContext ctx, 
80          final ChannelStateEvent cse) throws Exception {
81          final Channel ch = cse.getChannel();
82          LOG.debug("New CONNECT channel opened from proxy to web: {}", ch);
83          
84          // Could be null for testing.
85          if (this.channelGroup != null) {
86              this.channelGroup.add(ch);
87          }
88      }
89  
90      @Override
91      public void channelClosed(final ChannelHandlerContext ctx, 
92          final ChannelStateEvent e) throws Exception {
93          LOG.debug("Got closed event on proxy -> web connection: {}", 
94              e.getChannel());
95          this.netty4RelayChannel.close();
96      }
97  
98      @Override
99      public void exceptionCaught(final ChannelHandlerContext ctx, 
100         final ExceptionEvent e) throws Exception {
101         LOG.debug("Caught exception on proxy -> web connection: "+
102             e.getChannel(), e.getCause());
103         ProxyUtils.closeOnFlush(e.getChannel());
104     }
105 }
