52      public static void genericWrite(final Object message,
53          final ChannelFuture future) {
54          final Channel ch = future.getChannel();
55          if (ch.isConnected()) {
56              ch.write(message);
57          } else {
58              future.addListener(new ChannelFutureListener() {
59                  @Override
60                  public void operationComplete(final ChannelFuture cf)
61                      throws Exception {
62                      if (cf.isSuccess()) {
63                          ch.write(message);
