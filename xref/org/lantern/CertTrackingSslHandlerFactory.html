38  /**
39   * This class handles intercepting incoming SSL connections to the HTTP proxy
40   * server, associating incoming client certificates with their associated
41   * peers. Any incoming connections from non-trusted peers with non-trusted
42   * certificates will be rejected.

80      /**
81       * This method is called for every new pipeline that's created -- i.e.
82       * for every new incoming connection to the server. We need to reload
83       * the trust store each time to make sure we take into account all the last
84       * certificates from peers.
85       */
86      @Override
87      public HandshakeHandler newHandshakeHandler() {
88          log.debug("Creating new handshake handler...");
89          
90          final CertTrackingTrustManager certTracker = 
91              new CertTrackingTrustManager();
92          final SSLEngine engine = newSslEngine(certTracker);
93  
94          final SslHandlerInterceptor handler = new SslHandlerInterceptor(engine);
95          certTracker.setSslHandler(handler);
96          return new SslHandshakeHandler("ssl", handler);
97      }

99      public SSLEngine newSslEngine(final TrustManager trustManager) {
100         if (LanternUtils.isFallbackProxy()) {
101             return fallbackProxySslEngine();
102         } else {
103             return standardSslEngine(trustManager);
104         }

107     private SSLEngine fallbackProxySslEngine() {
108         log.debug("Using fallback proxy context");
109         final String PASS = "Be Your Own Lantern";
110         try {
111             final KeyStore ks = KeyStore.getInstance("JKS");
112 
113             final File keystore = new File(LanternUtils.getKeystorePath());
114             final InputStream is = new FileInputStream(keystore);
115             ks.load(is, PASS.toCharArray());
116 
117             // Set up key manager factory to use our key store
118             final KeyManagerFactory kmf = 
119                 KeyManagerFactory.getInstance("SunX509");
120             kmf.init(ks, PASS.toCharArray());
121 
122             // Initialize the SSLContext to work with our key managers.
123             final SSLContext serverContext = SSLContext.getInstance("TLS");
124             
125             // NO CLIENT AUTH!!
126             serverContext.init(kmf.getKeyManagers(), null, null);
127             final SSLEngine engine = serverContext.createSSLEngine();
128             engine.setUseClientMode(false);
129             return engine;
130         } catch (final Exception e) {
131             throw new Error(
132                     "Failed to initialize the server-side SSLContext", e);
133         }
134     }

136     private SSLEngine standardSslEngine(final TrustManager trustManager) {
137         log.debug("Using standard SSL context");
138         try {
139             final SSLContext context = SSLContext.getInstance("TLS");
140             context.init(this.keyStoreManager.getKeyManagerFactory().getKeyManagers(), 
141                 new TrustManager[]{trustManager}, null);
142             final SSLEngine engine = context.createSSLEngine();
143             engine.setUseClientMode(false);
144             engine.setNeedClientAuth(true);
145             final String[] suites = IceConfig.getCipherSuites();
146             if (suites != null && suites.length > 0) {
147                 engine.setEnabledCipherSuites(suites);
148             } else {
149                 // Can be null in tests.
150                 log.warn("No cipher suites?");
151             }
152             return engine;
153         } catch (final Exception e) {
154             throw new Error(
155                     "Failed to initialize the client-side SSLContext", e);
156         }

159     private class CertTrackingTrustManager implements X509TrustManager {
160 
161         private final Logger loggger = LoggerFactory.getLogger(getClass());
162         
163         private SslHandlerInterceptor handler;
164 
165         public void setSslHandler(final SslHandlerInterceptor handler) {

169         @Override
170         public void checkClientTrusted(final X509Certificate[] chain, String arg1)
171                 throws CertificateException {
172             loggger.debug("Checking client trusted...");
173             final X509Certificate cert = chain[0];
174             if (!LanternUtils.isFallbackProxy() && 
175                 !trustStore.containsCertificate(cert)) {
176                 loggger.warn("Certificate is not trusted!!");
177                 throw new CertificateException("not trusted");
178             }
179             
180             loggger.debug("Certificate trusted");
181             
182             Events.asyncEventBus().post(
183                 new IncomingPeerEvent(handler.channel, handler.trafficCounter, cert));
184             // We should already know about the peer at this point, and it's just
185             // a matter of correlating that peer with this certificate and 
186             // connection.
187             
188         }
189 
190         @Override
191         public void checkServerTrusted(final X509Certificate[] chain, String arg1)
192                 throws CertificateException {
193             throw new CertificateException(
194                     "Should never be checking server trust from the server");
195         }
196 
197         @Override
198         public X509Certificate[] getAcceptedIssuers() {
199             // We don't accept any issuers.
200             return new X509Certificate[]{};
201         }

204     private final class SslHandlerInterceptor extends SslHandler {
205 
206         private Channel channel;
207         private Netty3LanternTrafficCounterHandler trafficCounter;
208 
209         public SslHandlerInterceptor(final SSLEngine engine) {
210             super(engine);
211         }
212         
213         @Override
214         public void channelConnected(final ChannelHandlerContext ctx, 
215             final ChannelStateEvent e) throws Exception {
216             
217             log.debug("Got channel connected...");
218             try {
219                 
220                 // We basically want to add separate traffic handlers per IP, 
221                 // and we do that here. We have a new incoming socket and 
222                 // check for an existing handler. If it's there, we use it. 
223                 // Otherwise we add and use a new one.
224                 final InetSocketAddress isa = 
225                     (InetSocketAddress) ctx.getChannel().getRemoteAddress();
226                 final InetAddress address = isa.getAddress();
227                 
228                 final Netty3LanternTrafficCounterHandler newHandler = 
229                         new Netty3LanternTrafficCounterHandler(timer);
230                 final Netty3LanternTrafficCounterHandler existingHandler =
231                         handlers.putIfAbsent(address, newHandler);
232                 
233                 final Netty3LanternTrafficCounterHandler toUse;
234                 if (existingHandler == null) {
235                     toUse = newHandler;
236                 } else {
237                     log.debug("Using existing traffic counter...");
238                     toUse = existingHandler;
239                 }
240                 toUse.incrementSockets();
241                 this.channel = ctx.getChannel();
242                 this.trafficCounter = toUse;
243                 this.channel.getPipeline().addFirst(PIPELINE_ID, toUse);
244             } finally {
245                 // The message is then just passed to the next handler
246                 super.channelConnected(ctx, e);
247             }
248         }
