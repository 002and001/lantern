204     private void launch(final String... args) {
205         LOG.info("Starting Lantern...");
206         final Stopwatch earlyWatch = 
207             StopwatchManager.getStopwatch("pre-instance-creation", 
208                 STOPWATCH_LOG, STOPWATCH_GROUP);
209         earlyWatch.start();
210         // first apply any command line settings
211         final Options options = buildOptions();
212 
213         final CommandLineParser parser = new PosixParser();
214         final CommandLine cmd;
215         try {
216             cmd = parser.parse(options, args);
217             if (cmd.getArgs().length > 0) {
218                 throw new UnrecognizedOptionException("Extra arguments were provided");
219             }
220         }
221         catch (final ParseException e) {
222             printHelp(options, e.getMessage()+" args: "+Arrays.asList(args));
223             return;
224         }
225 
226         if (cmd.hasOption(OPTION_HELP)) {
227             printHelp(options, null);
228             return;
229         } else if (cmd.hasOption(OPTION_VERSION)) {
230             printVersion();
231             return;
232         }
233         earlyWatch.stop();
234         
235         final Stopwatch injectorWatch = 
236             StopwatchManager.getStopwatch("Guice-Injector", 
237                 STOPWATCH_LOG, STOPWATCH_GROUP);
238         injectorWatch.start();
239         injector = Guice.createInjector(this.lanternModule);
240         injectorWatch.stop();
241         
242         LOG.debug("Creating display...");
243 
244         final Stopwatch preInstanceWatch = 
245             StopwatchManager.getStopwatch("Pre-Instance-Creation", 
246                 STOPWATCH_LOG, STOPWATCH_GROUP);
247         preInstanceWatch.start();
248         // There are four cases here:
249         // 1) We're just starting normally
250         // 2) We're running with UI disabled (such as from a server), in
251         //    which case we don't show any UI elements
252         // 3) We're running on system startup (specified with --launchd flag)
253         //    and setup is not complete, in which case we show no splash screen,
254         //    but do show the UI at whatever setup step it's currently at
255         //    and put the app in the system tray
256         // 4) We're running on system startup (specified with --launchd flag)
257         //    and setup IS complete, in which case we show no splash screen,
258         //    do not show the UI, but do put the app in the system tray.
259         final boolean uiDisabled = cmd.hasOption(OPTION_DISABLE_UI);
260         final boolean launchD = cmd.hasOption(OPTION_LAUNCHD);
261 
262         configureCipherSuites();
263         final Display display;
264         if (uiDisabled) {
265             display = null;
266             preInstanceWatch.stop();
267         } else {
268             Display.setAppName("Lantern Beta");
269             display = DisplayWrapper.getDisplay();
270             preInstanceWatch.stop();
271             // Never show the splash screen on startup, even if setup is
272             // not complete.
273             if (!launchD) {
274                 /*
275                 splashScreen = instance(SplashScreen.class);
276                 final Stopwatch splashWatch = 
277                         StopwatchManager.getStopwatch("Splash-Screen-Init", 
278                             STOPWATCH_LOG, STOPWATCH_GROUP);
279                 splashWatch.start();
280                 splashScreen.init(display);
281                 splashWatch.stop();
282                 */
283             }
284         }
285 
286         model = instance(Model.class);
287         set = model.getSettings();
288         set.setUiEnabled(!uiDisabled);
289         instance(Censored.class);
290 
291         messageService = instance(MessageService.class);
292 
293         if (SystemUtils.IS_OS_MAC_OSX) {
294             final boolean sixtyFourBits =
295                 System.getProperty("sun.arch.data.model").equals("64");
296             if (!sixtyFourBits) {
297                 messageService.showMessage("Operating System Error",
298                         "We're sorry but Lantern requires a 64 bit operating " +
299                         "system on OSX! Exiting");
300                 
301                 System.exit(0);
302             }
303         }
304         jettyLauncher = instance(JettyLauncher.class);
305         
306         final Stopwatch jettyWatch = 
307                 StopwatchManager.getStopwatch("Jetty-Start", 
308                     STOPWATCH_LOG, STOPWATCH_GROUP);
309         jettyWatch.start();
310         jettyLauncher.start();
311         jettyWatch.stop();
312         modelUtils = instance(ModelUtils.class);
313         final boolean showDashboard = 
314                 shouldShowDashboard(model, uiDisabled, launchD);
315         if (showDashboard) {
316             browserService = instance(BrowserService.class);
317         }
318         launchLantern(showDashboard);
319         
320         keyStoreManager = instance(LanternKeyStoreManager.class);
321         instance(NatPmpService.class);
322         instance(UpnpService.class);
323         instance(LanternTrustStore.class);
324         instance(Proxifier.class);
325         final boolean showTray = !uiDisabled;
326 
327         if (showTray) {
328             systemTray = instance(SystemTray.class);
329             try {
330                 systemTray.start();
331             } catch (final Exception e) {
332                 LOG.error("Error starting tray?", e);
333             }
334         }
335 
336         xmpp = instance(DefaultXmppHandler.class);
337 
338         localCipherProvider = instance(LocalCipherProvider.class);
339 
340         internalState = instance(InternalState.class);
341         httpClientFactory = instance(HttpClientFactory.class);
342         syncService = instance(SyncService.class);
343 
344         proxyTracker = instance(ProxyTracker.class);
345 
346         LOG.debug("Processing command line options...");
347         processCommandLineOptions(cmd);
348         LOG.debug("Processed command line options...");
349 
350         instance(GeoIp.class);
351         statsUpdater = instance(StatsUpdater.class);
352 
353         model.getConnectivity().setInternet(false);
354         
355         // Use our stored STUN servers if available.
356         final Collection<String> stunServers = set.getStunServers();
357         if (stunServers != null && !stunServers.isEmpty()) {
358             LOG.info("Using stored STUN servers: {}", stunServers);
359             StunServerRepository.setStunServers(toSocketAddresses(stunServers));
360         }
361         
362         // Set up the give and get mode proxies
363         getModeProxy = instance(GetModeProxy.class);
364         giveModeProxy = instance(GiveModeProxy.class);
365         
366         startServices();
367 
368 
369         // This is necessary to keep the tray/menu item up in the case
370         // where we're not launching a browser.
371         // OX: YourKit was reporting deadlocks here.  This seems like a
372         // potentially expensive busy loop
373         if (display != null) {
374             while (!display.isDisposed ()) {
375                 if (!display.readAndDispatch ()) display.sleep ();
376             }
377         }
378     }

381      * This starts all of the services on a separate thread to avoid holding
382      * up the main thread that is in charge of displaying the UI.

384     private void startServices() {
385         final Thread t = new Thread(new Runnable() {
386 
387             @Override
388             public void run() {
389                 keyStoreManager.start();
390                 final ConnectivityChecker connectivityChecker =
391                     instance(ConnectivityChecker.class);
392                 final Timer timer = new Timer();
393                 timer.schedule(connectivityChecker, 0, 60 * 1000);
394 
395                 shutdownable(ModelIo.class);
396                 
397                 
398                 try {
399                     proxyTracker.start();
400                 } catch (final Exception e) {
401                     LOG.error("Could not start proxy tracker?", e);
402                 }
403                 xmpp.start();
404                 getModeProxy.start();
405                 giveModeProxy.start();
406 
407                 syncService.start();
408                 statsUpdater.start();
409 
410                 gnomeAutoStart();
411                 
412                 autoConnect();
413             }
414             
415         }, "Launcher-Start-Thread");
416         t.setDaemon(true);
417         t.start();
418     }

442     private <T> T instance(final Class<T> clazz) {
443         final String name = clazz.getSimpleName();
444         
445         final Stopwatch watch = 
446             StopwatchManager.getStopwatch(name, STOPWATCH_LOG, STOPWATCH_GROUP);
447         
448         watch.start();
449         
450         LOG.debug("Loading {}", name);
451         final T inst = injector.getInstance(clazz);
452         if (Shutdownable.class.isAssignableFrom(clazz)) {
453             addShutdownHook((Shutdownable) inst);
454         }
455         if (inst == null) {
456             LOG.error("Could not load instance of "+clazz);
457             throw new NullPointerException("Could not load instance of "+clazz);
458         }
459         /*
460         if (splashScreen != null) {
461             splashScreen.advanceBar();
462         }
463         */
464         watch.stop();
465         return inst;
466     }

480     private void gnomeAutoStart() {
481         // Before setup we should just do the default, which is to run on
482         // startup. The user can configure this differently at any point
483         // hereafter.
484         if (!SystemUtils.IS_OS_LINUX) {
485             return;
486         }
487         if (!LanternClientConstants.GNOME_AUTOSTART.isFile()) {
488             final File lanternDesktop;
489             final File candidate1 =
490                 new File(LanternClientConstants.GNOME_AUTOSTART.getName());
491             final File candidate2 =
492                 new File("install/linux", LanternClientConstants.GNOME_AUTOSTART.getName());
493             if (candidate1.isFile()) {
494                 lanternDesktop = candidate1;
495             } else if (candidate2.isFile()){
496                 lanternDesktop = candidate2;
497             } else {
498                 LOG.error("Could not find lantern.desktop file");
499                 return;
500             }
501             try {
502                 final File parent = LanternClientConstants.GNOME_AUTOSTART.getParentFile();
503                 if (!parent.isDirectory()) {
504                     if (!parent.mkdirs()) {
505                         LOG.error("Could not make dir for gnome autostart: "+parent);
506                         return;
507                     }
508                 }
509                 FileUtils.copyFileToDirectory(lanternDesktop, parent);
510 
511                 LOG.info("Copied {} to {}", lanternDesktop, parent);
512             } catch (final IOException e) {
513                 LOG.error("Could not configure gnome autostart", e);
514             }
515         }
516 
517         // Make sure our launcher entry has the Lantern icon, as opposed to
518         // the stock Chrome one.
519         LanternUtils.addStartupWMClass(
520                 "/usr/share/applications/lantern.desktop");
521     }

533     public static void configureCipherSuites() {
534         Security.addProvider(new BouncyCastleProvider());
535         if (!LanternUtils.isUnlimitedKeyStrength()) {
536             if (LanternUtils.isDevMode()) {
537                 System.err.println("PLEASE INSTALL UNLIMITED STRENGTH POLICY FILES WITH ONE OF THE FOLLOWING:\n" +
538                     "sudo cp install/java7/* $JAVA_HOME/jre/lib/security/\n" +
539                     "sudo cp install/java6/* $JAVA_HOME/jre/lib/security/\n" +
540                     "depending on the JVM you're running with. You may want to backup $JAVA_HOME/jre/lib/security as well.\n" +
541                     "JAVA_HOME is currently: "+System.getenv("JAVA_HOME"));
542                 
543                 // Don't exit if we're running on CI...
544                 final String env = System.getenv("BAMBOO");
545                 System.err.println("Env: "+System.getenv());
546                 if (!"true".equalsIgnoreCase(env)) {
547                     System.exit(1);
548                 }
549             }
550             if (!SystemUtils.IS_OS_WINDOWS_VISTA) {
551                 log("No policy files on non-Vista machine!!");
552             }
553             log("Reverting to weaker ciphers on Vista");
554             log("Look in "+ new File(SystemUtils.JAVA_HOME, "lib/security").getAbsolutePath());
555             IceConfig.setCipherSuites(new String[] {
556                     CIPHER_SUITE_LOW_BIT
557                 //"TLS_ECDHE_RSA_WITH_AES_128_CBC_SHA"
558                 //"TLS_ECDHE_RSA_WITH_RC4_128_SHA"
559             });
560         } else {
561             // Note the following just sets what cipher suite the server
562             // side selects. DHE is for perfect forward secrecy.
563 
564             // We include 128 because we never have enough permissions to
565             // copy the unlimited strength policy files on Vista, so we have
566             // to revert back to 128.
567             IceConfig.setCipherSuites(new String[] {
568                     CIPHER_SUITE_LOW_BIT,
569                     CIPHER_SUITE_HIGH_BIT
570                 //"TLS_ECDHE_RSA_WITH_AES_256_CBC_SHA",
571                 //"TLS_DHE_RSA_WITH_AES_128_CBC_SHA"
572                 //"TLS_RSA_WITH_RC4_128_SHA"
573                 //"TLS_ECDHE_RSA_WITH_RC4_128_SHA"
574             });
575         }
576     }

587     private static Collection<InetSocketAddress> toSocketAddresses(
588         final Collection<String> stunServers) {
589         final Collection<InetSocketAddress> isas =
590             new HashSet<InetSocketAddress>();
591         for (final String server : stunServers) {
592             final String host = StringUtils.substringBefore(server, ":");
593             final String port = StringUtils.substringAfter(server, ":");
594             isas.add(new InetSocketAddress(host, Integer.parseInt(port)));
595         }
596         return isas;
597     }

610     private void loadLocalPasswordFile(final String pwFilename) {
611         //final LocalCipherProvider lcp = localCipherProvider;
612         if (!localCipherProvider.requiresAdditionalUserInput()) {
613             LOG.error("Settings do not require a password to unlock.");
614             System.exit(1);
615         }
616 
617         if (StringUtils.isBlank(pwFilename)) {
618             LOG.error("No filename specified to --{}", OPTION_PASSWORD_FILE);
619             System.exit(1);
620         }
621         final File pwFile = new File(pwFilename);
622         if (!(pwFile.exists() && pwFile.canRead())) {
623             LOG.error("Unable to read password from {}", pwFilename);
624             System.exit(1);
625         }
626 
627         LOG.info("Reading local password from file \"{}\"", pwFilename);
628         try {
629             final String pw = FileUtils.readLines(pwFile, "US-ASCII").get(0);
630             final boolean init = !localCipherProvider.isInitialized();
631             localCipherProvider.feedUserInput(pw.toCharArray(), init);
632         }
633         catch (final IndexOutOfBoundsException e) {
634             LOG.error("Password in file \"{}\" was incorrect", pwFilename);
635             System.exit(1);
636         }
637         catch (final InvalidKeyException e) {
638             LOG.error("Password in file \"{}\" was incorrect", pwFilename);
639             System.exit(1);
640         }
641         catch (final GeneralSecurityException e) {
642             LOG.error("Failed to initialize using password in file \"{}\": {}", pwFilename, e);
643             System.exit(1);
644         }
645         catch (final IOException e) {
646             LOG.error("Failed to initialize using password in file \"{}\": {}", pwFilename, e);
647             System.exit(1);
648         }
649     }

681     private void autoConnect() {
682         LOG.debug("Connecting if oauth is configured...");
683         // This won't connect in the case where the user hasn't entered
684         // their user name and password and the user is running with a UI.
685         // Otherwise, it will connect.
686         if (modelUtils.isConfigured()) {
687             final Runnable runner = new Runnable() {
688                 @Override
689                 public void run() {
690                     try {
691                         xmpp.connect();
692                         if (model.getModal() == Modal.connecting) {
693                             internalState.advanceModal(null);
694                         }
695                     } catch (final IOException e) {
696                         LOG.debug("Could not login", e);
697                     } catch (final CredentialException e) {

699                     } catch (final NotInClosedBetaException e) {

701                     }
702                 }
703             };
704             final Thread t = new Thread(runner, "Auto-Starting-Thread");
705             t.setDaemon(true);
706             t.start();
707         } else {
708             LOG.debug("Not auto-logging in with model:\n{}", model);
709             if (model.isSetupComplete())
710                 Events.syncModal(model, Modal.authorize);
711         }

823     private void handleError(final Throwable t, final boolean exit) {
824         final String msg = msg(t);
825         LOG.error("Uncaught exception on" +
826                 "\nOS_NAME: "+SystemUtils.OS_NAME +
827                 "\nOS_ARCH: "+SystemUtils.OS_ARCH +
828                 "\nOS_VERSION: "+SystemUtils.OS_VERSION +
829                 "\nUSER_COUNTRY: "+SystemUtils.USER_COUNTRY +
830                 "\nUSER_LANGUAGE: "+SystemUtils.USER_LANGUAGE +
831                 "\n\n"+msg, t);
832         if (t instanceof SWTError || msg.contains("SWTError")) {
833             System.out.println(
834                 "To run without a UI, run lantern with the --" +
835                 OPTION_DISABLE_UI +
836                 " command line argument");
837         }
838         else if (t instanceof UnsatisfiedLinkError &&
839             msg.contains("Cannot load 32-bit SWT libraries on 64-bit JVM")) {
840             messageService.showMessage("Architecture Error",
841                 "We're sorry, but it appears you're running 32-bit Lantern on a 64-bit JVM.");
842         }
843         else if (!lanternStarted && set != null && set.isUiEnabled()) {
844             LOG.info("Showing error to user...");
845             messageService.showMessage("Startup Error",
846                "We're sorry, but there was an error starting Lantern " +
847                "described as '"+msg+"'.");
848         }
849         if (exit) {
850             LOG.info("Exiting Lantern");
851             // Give the logger a second to report the error.
852             try {Thread.sleep(6000);} catch (final InterruptedException e) {}
853             System.exit(1);
854         }
855     }

974                 cmd.getOptionValue(secOpt));

976 
977         final String credOpt = OPTION_OAUTH2_USER_CREDENTIALS_FILE;
978         if (cmd.hasOption(credOpt)) {
979             modelUtils.loadOAuth2UserCredentialsFile(
980                 cmd.getOptionValue(credOpt));
981         }

983         //final Settings set = LanternHub.settings();
984 
985         set.setUseTrustedPeers(parseOptionDefaultTrue(cmd, OPTION_TRUSTED_PEERS));
986         set.setUseAnonymousPeers(parseOptionDefaultTrue(cmd, OPTION_ANON_PEERS));
987         set.setUseLaeProxies(parseOptionDefaultTrue(cmd, OPTION_LAE));
988         set.setUseCentralProxies(parseOptionDefaultTrue(cmd, OPTION_CENTRAL));
989         set.setUdpProxyPriority(cmd.getOptionValue(OPTION_UDP_PROXY_PRIORITY, "lower").toUpperCase());
990         
991         final boolean tcp = parseOptionDefaultTrue(cmd, OPTION_TCP);
992         final boolean udp = parseOptionDefaultTrue(cmd, OPTION_UDP);
993         IceConfig.setTcp(tcp);
994         IceConfig.setUdp(udp);
995         set.setTcp(tcp);
996         set.setUdp(udp);

1014         if (cmd.hasOption(OPTION_DISABLE_KEYCHAIN)) {

1027             set.setBindToLocalhost(false);
1028         }
1029 
1030         LOG.info("Running API on port: {}", StaticSettings.getApiPort());
1031 
1032         if (cmd.hasOption(OPTION_SERVER_PORT)) {
1033             final String serverPortStr =
1034                 cmd.getOptionValue(OPTION_SERVER_PORT);
1035             LOG.debug("Using command-line proxy port: "+serverPortStr);
1036             final int serverPort = Integer.parseInt(serverPortStr);
1037             set.setServerPort(serverPort);
1038         } else {

1056         } else if (cmd.hasOption(OPTION_GET)) {
1057             model.getSettings().setMode(Mode.get);
