44  /**
45   * Compresses the MaxMind GeoLite database for Lantern. This compression method
46   * is optimized for Lantern-UI use; much data is discarded.
47   *
48   * Only countries and screen-resolution lat/lon are preserved.
49   *
50   * The data must be decompressed before use.
51   */
52  
53  public class GeoIpCompressor {
54      private static final Logger LOG = LoggerFactory
55              .getLogger(GeoIpCompressor.class);

57      public static final int SCREEN_WIDTH = 960;
58      public static final int SCREEN_HEIGHT = 625;
59  
60      // the quantized lat/lon is stored as lon + lat * SCREEN_WIDTH
61      IntList pixelIdToQuantizedLatLon = new IntArrayList();
62      IntList pixelIdToCountry = new IntArrayList();
63  
64      Map<String, Integer> countryToCountryId = new HashMap<String, Integer>();
65      List<String> countryIdToCountry = new ArrayList<String>();
66  
67      // there is a relatively small number of pixels that cities quantize to,
68      // so we'll give each one an id. This is the internal ID (and index
69      // into the pixelIdToQuantiedLatLon list)
70      Map<Pair<Integer, Integer>, Integer> quantizedLocationToPixelId = new HashMap<Pair<Integer, Integer>, Integer>();
71  
72      // maps from the database's location id to our pixel id
73      Map<Integer, Integer> locIdToPixelId = new HashMap<Integer, Integer>();
74  
75      Multiset<Integer> rangeCounts = HashMultiset.create();
76  
77      final IntList ipRangeList = new IntArrayList();
78      final IntList pixelIdList = new IntArrayList();

83      /**
84       * Returns the latitude and longitude, quantized to screen resolution
85       * (960x625)
86       *
87       * @return
88       */
89      public int quantizedLatLon(double lat, double lon) {
90          int quantizedLon = (int) ((lon + 180) * SCREEN_WIDTH / 360);
91          int quantizedLat = (int) ((lat + 90) * SCREEN_HEIGHT / 180);
92  
93          return quantizedLat * SCREEN_WIDTH + quantizedLon;
94      }
95  
96      public double getLatFromQuantized(int quantized) {
97          int scaledLat = quantized / SCREEN_WIDTH;
98          return ((float) scaledLat) * 180 / SCREEN_HEIGHT - 90;
99      }
100 
101     public double getLonFromQuantized(int quantized) {
102         int scaledLon = quantized % SCREEN_WIDTH;
103         return ((float) scaledLon) * 360 / SCREEN_WIDTH - 180;
104     }

151     private void writeDecompressedBlocks(File blockFile) throws IOException {
152         CsvWriter writer = new CsvWriter(blockFile.getAbsolutePath());
153 
154         writer.writeComment("Derived from MaxMind GeoLiteCity.  Copyright (c) 2012 MaxMind LLC.");
155         writer.writeRecord(new String[] { "startIpNum", "endIpNum", "pixelId" });
156 
157         long start = 0;
158         for (int i = 0; i < ipRangeList.size(); ++i) {
159             int ipRange = ipRangeList.get(i);
160             int pixelId = pixelIdList.get(i);
161             long end = start + ipRange - 1;
162             writer.writeRecord(new String[] { "" + start, "" + end,
163                     "" + pixelId });
164             start = end + 1;
165         }
166         writer.close();
167

170     private void readCompressedBlocks(InputStream inStream) throws IOException {
171 
172         // the range sizes which only appear once
173         IntList singletons = new IntArrayList();
174 
175         int count = 0;
176         while (true) {
177             if (count++ > 500000) {
178                 throw new RuntimeException("Unexpectedly large number of singleton locations (probably a corrupt geoip.db)");
179             }
180             byte[] b = new byte[4];
181             inStream.read(b);
182             int i = BitUtils.byteArrayToInteger(b);
183             if (i == -1)
184                 break;
185             singletons.add(i);
186         }
187 
188         IntList rangeSizes = new IntArrayList();
189 
190         rangeSizes.add(0); // special case for singletons
191         while (true) {
192             byte[] b = new byte[4];
193             inStream.read(b);
194             int i = BitUtils.byteArrayToInteger(b);
195             if (i == -1)
196                 break;
197             rangeSizes.add(i);
198         }
199 
200         int singleton = 0;
201 
202         // now read in the compressed blocks
203         InputStream shielded = new CloseShieldInputStream(inStream);
204 
205         RCModel model = new FenwickTreeModel(rangeSizes.size());
206         DecompressedInputStream compressedStream = new DecompressedInputStream(
207                 shielded, model);
208 
209         while (true) {
210             int i = compressedStream.read();
211             if (i == -1) {
212                 break;
213             }
214             int range = rangeSizes.get(i);
215             if (range == 0) {
216                 range = singletons.get(singleton++);
217             }
218             ipRangeList.add(range);
219         }
220         compressedStream.close();
221 
222         // and the locations-for-blocks
223         shielded = new CloseShieldInputStream(inStream);
224 
225         // Order1Model model2 = new Order1Model(pixelIdToQuantizedLatLon.size(),
226         // pixelIdToCountry, countryIdToCountry.size());
227         RCModel model2 = new FenwickTreeModel(pixelIdToQuantizedLatLon.size());
228         DecompressedInputStream compressedStream2 = new DecompressedInputStream(
229                 shielded, model2);
230 
231         // model2.setPrevious(pixelIdToCountry.get(compressedStream2._nextByte));
232         while (true) {
233             int pixelId = compressedStream2.read();
234             if (pixelId == -1) {
235                 break;
236             }
237             pixelIdList.add(pixelId);
238             /*
239              * if (compressedStream2._nextByte == -1) break;
240              * model2.setPrevious(pixelIdToCountry
241              * .get(compressedStream2._nextByte));
242              */
243         }
244         compressedStream2.close();
245         assert pixelIdList.size() == ipRangeList.size();
246     }

248     private void writeDecompressedLocations(File locationFile)
249             throws IOException {
250         CsvWriter writer = new CsvWriter(locationFile.getAbsolutePath());
251 
252         writer.writeComment("Derived from MaxMind GeoLiteCity.  Copyright (c) 2012 MaxMind LLC.");
253         writer.writeRecord(new String [] {"pixelId","country","latitude","longitude"});
254 
255         for (int i = 0; i < pixelIdToCountry.size(); ++i) {
256             int countryId = pixelIdToCountry.get(i);
257             String country = countryIdToCountry.get(countryId);
258             int quantized = pixelIdToQuantizedLatLon.get(i);
259             double latitude = getLatFromQuantized(quantized);
260             double longitude = getLonFromQuantized(quantized);
261             writer.writeRecord(new String[] { "" + i, country, "" + latitude,
262                     "" + longitude });
263         }
264         writer.close();
265     }

267     private void readCompressedLocations(InputStream inStream)
268             throws IOException {
269         byte[] b = new byte[4];
270         inStream.read(b);
271         int numPixelIds = BitUtils.byteArrayToInteger(b);
272         LOG.debug("nPixelIds = " + numPixelIds);
273         BitInputStream bitStream = new BitInputStream(inStream);
274 
275         for (int i = 0; i < numPixelIds; ++i) {
276             pixelIdToQuantizedLatLon.add(bitStream.read(20));
277         }
278         bitStream.flush();
279 
280         // read in the list of countries ordered by id
281         inStream.read(b);
282         int nCountries = BitUtils.byteArrayToInteger(b);
283         LOG.debug("nCountries = " + nCountries);
284         for (int i = 0; i < nCountries; ++i) {
285             byte[] countryCodeBytes = new byte[2];
286             inStream.read(countryCodeBytes);
287             String country = new String(countryCodeBytes);
288             countryIdToCountry.add(country);
289         }
290 
291         // read in the country for each pixel id
292         RCModel model = new FenwickTreeModel(nCountries);
293 
294         InputStream shielded = new CloseShieldInputStream(inStream);
295         InputStream compressedStream = new DecompressedInputStream(shielded,
296                 model);
297 
298         while (true) {
299             int read = compressedStream.read();
300             if (read == -1) {
301                 break;
302             }
303             pixelIdToCountry.add(read);
304         }
305         compressedStream.close();
306 
307     }

309     private void compressInternal(File in, File out) throws IOException {
310         File locationFile = new File(in, "GeoLiteCity-Location.csv");
311 
312         readLocations(locationFile);
313 
314         CountingOutputStream outStream = new CountingOutputStream(
315                 new BufferedOutputStream(new FileOutputStream(out)));
316 
317         writeCompressedLocations(outStream);
318 
319         LOG.debug("Bytes used after writing locations: "
320                 + outStream.getByteCount());
321 
322         File blocksFile = new File(in, "GeoLiteCity-Blocks.csv");
323         readBlocks(blocksFile);
324         writeCompressedBlocks(outStream);
325         outStream.close();
326 
327         LOG.debug("Done writing compressed GeoIp file");
328     }

330     private void readBlocks(File blocksFile) throws IOException {
331         CsvReader reader = new CsvReader(blocksFile.getAbsolutePath(), ',',
332                 Charset.forName("UTF8"));
333         reader.setTextQualifier('"');
334         reader.skipLine();
335         reader.readHeaders();
336 
337         int lastPixelId = -1;
338         long lastStartIpNum = -1;
339 
340         long endIpNum = -1;
341         int pixelId;
342         while (reader.readRecord()) {
343             long startIpNum = Long.parseLong(reader.get("startIpNum"));
344             endIpNum = Long.parseLong(reader.get("endIpNum"));
345             int locId = Integer.parseInt(reader.get("locId"));
346 
347             pixelId = locIdToPixelId.get(locId);
348 
349             if (lastStartIpNum == -1) {
350                 // first range does not start at zero... but we'll pretend it
351                 // does
352                 lastStartIpNum = 0;
353                 lastPixelId = pixelId;
354                 continue;
355             }
356 
357             if (pixelId != lastPixelId) {
358                 // store the previous range
359                 int range = (int) (startIpNum - lastStartIpNum);
360                 ipRangeList.add(range);
361                 rangeCounts.add(range);
362                 pixelIdList.add(lastPixelId);
363 
364                 lastStartIpNum = startIpNum;
365                 lastPixelId = pixelId;
366             }
367         }
368         // handle the last range;
369         int range = (int) (endIpNum - lastStartIpNum + 1);
370         ipRangeList.add(range);
371         rangeCounts.add(range);
372         pixelIdList.add(lastPixelId);
373     }

375     private void writeCompressedLocations(CountingOutputStream outStream)
376             throws IOException {
377 
378         int numLocations = pixelIdToQuantizedLatLon.size();
379         LOG.debug(numLocations + " pixel ids");
380         outStream.write(BitUtils.toByteArray(numLocations));
381         BitOutputStream bitStream = new BitOutputStream(outStream);
382         for (int quantized : pixelIdToQuantizedLatLon) {
383             bitStream.write(quantized, 20);
384         }
385         bitStream.flush();
386 
387         LOG.debug("Bytes used after quantized: " + outStream.getByteCount());
388         outStream.write(BitUtils.toByteArray(countryIdToCountry.size()));
389         // write out the list of countries by id
390         for (String country : countryIdToCountry) {
391             outStream.write(country.getBytes());
392         }
393         LOG.debug("Bytes used after country: " + outStream.getByteCount());
394 
395         // write out the range-coded countries for pixel ids
396         RCModel model = new FenwickTreeModel(countryToCountryId.size());
397 
398         CloseShieldOutputStream shielded = new CloseShieldOutputStream(
399                 outStream);
400         CompressedOutputStream compressedStream = new CompressedOutputStream(
401                 shielded, model);
402 
403         for (int countryId : pixelIdToCountry) {
404             compressedStream.write(countryId);
405         }
406         compressedStream.close();
407 
408     }

410     private void writeCompressedBlocks(CountingOutputStream outStream)
411             throws IOException {
412         // write out the blocks themselves
413 
414         // first, let's write out the sizes of all of the ranges, in order of
415         // appearance
416 
417         final HashMap<Integer, Integer> rangeNumbering = new HashMap<Integer, Integer>();
418         for (int range : ipRangeList) {
419             if (rangeCounts.count(range) <= 2) {
420                 outStream.write(BitUtils.toByteArray(range));
421                 // all singletons are zeroes;
422                 // treat doubletons as singletons
423                 rangeNumbering.put(range, 0);
424             }
425         }
426         outStream.write(BitUtils.toByteArray(-1));
427 
428         LOG.debug("Bytes used after singletons/doubletons: "
429                 + outStream.getByteCount());
430         int number = 1;
431         for (int range : ipRangeList) {
432             if (rangeCounts.count(range) > 2) {
433                 if (!rangeNumbering.containsKey(range)) {
434                     rangeNumbering.put(range, number++);
435                     outStream.write(BitUtils.toByteArray(range));
436                 }
437             }
438         }
439         outStream.write(BitUtils.toByteArray(-1));
440         LOG.debug("Bytes used after other range sizes: "
441                 + outStream.getByteCount());
442 
443         CloseShieldOutputStream shielded = new CloseShieldOutputStream(
444                 outStream);
445 
446         // now write out the compressed blocks
447         RCModel model = new FenwickTreeModel(number);
448 
449         CompressedOutputStream compressedStream = new CompressedOutputStream(
450                 shielded, model);
451         for (int range : ipRangeList) {
452             compressedStream.write(rangeNumbering.get(range));
453         }
454         compressedStream.close();
455 
456         LOG.debug("Bytes used after iprangelist: " + outStream.getByteCount());
457 
458         shielded = new CloseShieldOutputStream(outStream);
459 
460         // Write the compressed locations for each block
461 
462         // This order-1 model conditions on the country of the previous
463         // location
464         /*
465          * Order1Model model2 = new Order1Model(pixelIdToCountry.size(),
466          * pixelIdToCountry, countryIdToCountry.size());
467          */
468         RCModel model2 = new FenwickTreeModel(pixelIdToCountry.size());
469         compressedStream = new CompressedOutputStream(shielded, model2);
470         for (int pixelId : pixelIdList) {
471             compressedStream.write(pixelId);
472         }
473         compressedStream.close();
474 
475         LOG.debug("Bytes used after pixelidlist: " + outStream.getByteCount());
476     }

478     /**
479      * read in the locations csv file and set up location mappings
480      *
481      * @param locationFile
482      */
483     private void readLocations(File locationFile) throws IOException {
484         CsvReader reader = new CsvReader(locationFile.getAbsolutePath(), ',',
485                 Charset.forName("UTF8"));
486         reader.skipLine();
487         reader.readHeaders();
488 
489         int nextPixelId = 0;
490         int nextCountryId = 0;
491 
492         while (reader.readRecord()) {
493             double lat = Double.parseDouble(reader.get("latitude"));
494             double lon = Double.parseDouble(reader.get("longitude"));
495             String country = reader.get("country");
496             country = normalizeCountry(country);
497 
498             Integer countryId = countryToCountryId.get(country);
499             if (countryId == null) {
500                 countryId = nextCountryId++;
501                 countryToCountryId.put(country, countryId);
502                 countryIdToCountry.add(country);
503             }
504 
505             int locId = Integer.parseInt(reader.get("locId"));
506 
507             int quantized = quantizedLatLon(lat, lon);
508             Pair<Integer, Integer> key = Pair.of(quantized, countryId);
509             Integer pixelId = quantizedLocationToPixelId.get(key);
510             if (pixelId == null) {
511                 pixelId = nextPixelId++;
512                 pixelIdToQuantizedLatLon.add(quantized);
513                 pixelIdToCountry.add(countryId);
514                 quantizedLocationToPixelId.put(key, pixelId);
515             }
516             locIdToPixelId.put(locId, pixelId);
517         }
518 
519     }

521     /** normalize all null islands to one place */
522     private String normalizeCountry(String country) {
523         if (country.equals("A1") || country.equals("A2")) {
524             country = "O1";
525         }
526         return country;
527     }
