32  /**
33   * This class manages the local trust store of external entities we trust
34   * for creating SSL connections. This includes both hard coded certificates
35   * for a small number of services we use as well as dynamically added
36   * certificates for peers. Peer certificates must be discovered and added
37   * through a trusted channel such as one of the hard coded trusted authorities
38   * (aka Google Talk with its equifax certificate as of this writing).
39   */
40  @Singleton
41  public class LanternTrustStore {
42  
43      private final static Logger log = 
44          LoggerFactory.getLogger(LanternTrustStore.class);
45      
46      private final AtomicReference<SSLContext> sslContextRef = 
47              new AtomicReference<SSLContext>();
48      private final LanternKeyStoreManager ksm;
49  
50      private final KeyStore trustStore;
51  
52      private TrustManagerFactory tmf;
53      
54      @Inject
55      public LanternTrustStore(final LanternKeyStoreManager ksm) {
56          this.ksm = ksm;
57          this.trustStore = blankTrustStore();
58          this.tmf = initTrustManagerFactory();
59      }
60  
61      /**
62       * Every time the trust store changes, we need to recreate particularly
63       * our SSL context because we now trust different servers. We reload only
64       * on trust store changes as an optimization to avoid loading the whole
65       * SSL context from scratch for each new outgoing socket.
66       */
67      private void onTrustStoreChanged() {
68          this.tmf = initTrustManagerFactory();
69          this.sslContextRef.set(provideSslContext());
70      }
71  
72      private TrustManagerFactory initTrustManagerFactory() {
73          try {
74              final TrustManagerFactory trustManagerFactory = 
75                      TrustManagerFactory.getInstance(
76                      TrustManagerFactory.getDefaultAlgorithm());
77              
78              // We create the trust manager factory with the latest trust store.
79              trustManagerFactory.init(this.trustStore);
80              return trustManagerFactory;
81          } catch (final NoSuchAlgorithmException e) {
82              log.error("Could not load algorithm?", e);
83              throw new Error("Could not load algorithm", e);
84          } catch (final KeyStoreException e) {
85              log.error("Could not load keystore?", e);
86              throw new Error("Could not load keystore for tmf", e);
87          }
88      }
89  
90      private KeyStore blankTrustStore() {
91          try {
92              final KeyStore ks = KeyStore.getInstance("JKS");
93              ks.load(null, null);
94              
95              final CertificateFactory cf = CertificateFactory.getInstance("X.509");
96              addCert(ks, cf, "littleproxy", LITTLEPROXY);
97              addCert(ks, cf, "digicerthighassurancerootca", DIGICERT);
98              addCert(ks, cf, "equifaxsecureca", EQUIFAX);
99              addCert(ks, cf, "equifaxsecureca2", EQUIFAX2);
100             return ks;
101         } catch (final KeyStoreException e) {
102             log.error("Could not load keystore?", e);
103             throw new Error("Could not load blank trust store", e);
104         } catch (final NoSuchAlgorithmException e) {
105             log.error("No such algo?", e);
106             throw new Error("Could not load blank trust store", e);
107         } catch (final CertificateException e) {
108             log.error("Bad cert?", e);
109             throw new Error("Could not load blank trust store", e);
110         } catch (final IOException e) {
111             log.error("Could not load?", e);
112             throw new Error("Could not load blank trust store", e);
113         } 
114     }
115 
116     private void addCert(final KeyStore ks, final CertificateFactory cf,
117         final String alias, final String pemCert) 
118                 throws CertificateException, KeyStoreException {
119         final InputStream bis = 
120             new ByteArrayInputStream(pemCert.getBytes(Charsets.UTF_8));
121         final Certificate cert = cf.generateCertificate(bis);
122         ks.setCertificateEntry(alias, cert);
123     }
124 
125     public void addBase64Cert(final URI jid, final String base64Cert) {
126         log.debug("Adding base 64 cert for {} to trust store", jid);
127         Events.asyncEventBus().post(new PeerCertEvent(jid, base64Cert));
128 
129         final byte[] decoded = Base64.decodeBase64(base64Cert);
130         final InputStream is = new ByteArrayInputStream(decoded);
131         addCert(jid.toASCIIString(), is);
132     }
133     
134     public void addCert(final String alias, final InputStream is) {
135         log.debug("Importing cert");
136         try {
137             final CertificateFactory cf = CertificateFactory.getInstance("X.509");
138             final Certificate certificate = cf.generateCertificate(is);
139             this.trustStore.setCertificateEntry(alias, certificate);
140         } catch (final CertificateException e) {
141             log.error("Could not load cert - cert error?", e);
142         } catch (final KeyStoreException e) {
143             log.error("Could not load cert into keystore?", e);
144         } finally {
145             IOUtils.closeQuietly(is);
146         }
147         
148         onTrustStoreChanged();
149     }

153     /**
154      * Accessor for the SSL context. This is regenerated whenever
155      * we receive new certificates.
156      * 
157      * @return The SSL context.
158      */
159     public synchronized SSLContext getSslContext() {
160         if (this.sslContextRef.get() == null) {
161             this.sslContextRef.set(provideSslContext());
162         }
163         return sslContextRef.get();
164     }
165     
166     /**
167      * Return an new/unused {@link SSLEngine} reflecting our {@link SSLContext}.
168      * For performance, these are created eagerly ahead of time.
169      */
170     public SSLEngine newSSLEngine() {
171         return getSslContext().createSSLEngine();
172     }
173 
174     private SSLContext provideSslContext() {
175         try {
176             final SSLContext context = SSLContext.getInstance("TLS");
177             
178             // Create the context with the latest trust manager factory.
179             context.init(this.ksm.getKeyManagerFactory().getKeyManagers(), 
180                 this.tmf.getTrustManagers(), null);
181             return context;
182         } catch (final Exception e) {
183             throw new Error(
184                     "Failed to initialize the client-side SSLContext", e);
185         }
186     }
187     
188     public void deleteCert(final String alias) {
189         try {
190             this.trustStore.deleteEntry(alias);
191             onTrustStoreChanged();
192         } catch (final KeyStoreException e) {
193             log.debug("Error removing entry -- doesn't exist?", e);
194         }
195     }
196 
197     /**
198      * Checks if the trust store contains exactly this certificate. This 
199      * doesn't worry about certificate chaining or anything like that -- 
200      * this trust store must instead contain the actual certificate.
201      * 
202      * @param cert The certificate to check.
203      * @return <code>true</code> if the trust store contains the certificate,
204      * otherwise <code>false</code>.
205      */
206     public boolean containsCertificate(final X509Certificate cert) {
207         log.debug("Loading trust store...");
208         
209         // We could use getCertificateAlias here, but that will iterate through
210         // everything, potentially causing issues when there are a lot of certs.
211         final String alias = 
212             cert.getIssuerDN().getName().substring(3).toLowerCase();
213         log.debug("Looking for alias {}", alias);
214         try {
215             if (log.isDebugEnabled()) {
216                 log.debug("All aliases");
217                 Enumeration<String> aliases = this.trustStore.aliases();
218                 while (aliases.hasMoreElements()) {
219                     log.debug(aliases.nextElement());
220                 }
221             }
222             final Certificate existingCert = this.trustStore.getCertificate(alias);
223             log.trace("Existing certificate: {}", (existingCert));
224             return existingCert != null && existingCert.equals(cert);
225         } catch (final KeyStoreException e) {
226             log.warn("Exception accessing keystore", e);
227             return false;
228         }
229     }
230     
231     private void listEntries(final KeyStore ks) {
232         try {
233             final Enumeration<String> aliases = ks.aliases();
234             while (aliases.hasMoreElements()) {
235                 final String alias = aliases.nextElement();
236                 //System.err.println(alias+": "+ks.getCertificate(alias));
237                 System.err.println(alias);
238                 log.debug(alias);
239             }
240         } catch (final KeyStoreException e) {
241             log.warn("KeyStore error", e);
242         }
243     }
244     
245     public void listEntries() {
246         listEntries(trustStore);
247     }
