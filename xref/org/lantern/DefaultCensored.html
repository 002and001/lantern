103     private boolean isCensored(final InetAddress address) {
104         return isMatch(address, CENSORED);
105     }
106     private boolean isMatch(final InetAddress address, 
107         final Collection<String> countries) { 
108         if (address == null) {
109             return true;
110         }
111         return countries.contains(countryCode(address));
112     }

114     private String countryCode(final InetAddress address) {
