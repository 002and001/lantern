39       * Map of kscope advertisements for which we are awaiting corresponding
40       * certificates.

42      private final ConcurrentHashMap<URI, LanternKscopeAdvertisement> awaitingCerts =

64      @Override
65      public boolean handleAd(final String from,
66              final LanternKscopeAdvertisement ad) {
67          // output a bell character to call more attention
68          log.debug("\u0007*** got kscope ad from {} for {}", from, ad.getJid());
69          Events.asyncEventBus().post(new KscopeAdEvent(ad));
70          final LanternKscopeAdvertisement existing =
71              awaitingCerts.put(LanternUtils.newURI(ad.getJid()), ad);
72  
73          //ignore kscope ads directly or indirectly from untrusted sources
74          //(they might have been relayed via untrusted sources in the middle,
75          //but there is nothing we can do about that)
76  
77          if (isUntrusted(ad.getJid())) {
78              return false;
79          }
80  
81          if (isUntrusted(from)) {
82              return false;
83          }
84  
85          if (existing != null) {
86              if (existing.equals(ad)) {
87                  log.debug("Ignoring identical kscope ad - already processed");
88                  return false;
89              }
90          }
91          // do we want to relay this?
92          int inboundTtl = ad.getTtl();
93          if(inboundTtl <= 0) {
94              log.debug("End of the line for kscope ad for {} from {}.", 
95                  ad.getJid(), from
96              );
97              return true;
98          }
99          TrustGraphNodeId nid = new BasicTrustGraphNodeId(ad.getJid());
100         TrustGraphNodeId nextNid = routingTable.getNextHop(nid);
101         if (nextNid == null) {
102             // This will happen when we're not connected to any other peers,
103             // for example.
104             log.debug("Could not relay ad: Node ID not in routing table");
105             return true;
106         }
107         LanternKscopeAdvertisement relayAd = 
108             LanternKscopeAdvertisement.makeRelayAd(ad);
109 
110         final String relayAdPayload = JsonUtils.jsonify(relayAd);
111         final BasicTrustGraphAdvertisement message =
112             new BasicTrustGraphAdvertisement(nextNid, relayAdPayload, 
113                 relayAd.getTtl()
114             );
115 
116         final TrustGraphNode tgn = 
117             new LanternTrustGraphNode(xmppHandler);
118         
119         tgn.sendAdvertisement(message, nextNid, relayAd.getTtl()); 
120         return true;
121     }

123     private boolean isUntrusted(final String jid) {
124         Friends friends = model.getFriends();
125         String email = LanternXmppUtils.jidToEmail(jid);
126         Friend friend = friends.get(email);
127         if (friend != null && friend.getStatus() == Status.rejected) {
128             return true;
129         }
130         return false;
131     }

133     @Override
134     public void onBase64Cert(final URI jid, final String base64Cert) {
135         this.trustStore.addBase64Cert(jid, base64Cert);
136         
137         final LanternKscopeAdvertisement ad = awaitingCerts.remove(jid);
138         if (ad != null) {
139             log.debug("Adding proxy...");
140             if (ad.hasMappedEndpoint()) {
141                 this.proxyTracker.addProxy(jid, 
142                         LanternUtils.isa(ad.getAddress(), ad.getPort()));
143             } else {
144                 this.proxyTracker.addJidProxy(LanternUtils.newURI(ad.getJid()));
145             }
146             
147             // Also add the local network advertisement in case they're on
148             // the local network.
149             this.proxyTracker.addProxy(jid, 
150                 LanternUtils.isa(ad.getLocalAddress(), ad.getLocalPort()));
151             processedAds.add(ad);
152         } else {
153             if (this.proxyTracker.hasJidProxy(jid)) {
154                 log.debug("Ignoring cert from peer we already have: {}", jid);
155                 return;
156             } else {
157                 // This could happen if we negotiated certs in some way other 
158                 // than in response to a kscope ad, such as for peers from the 
159                 // controller or just peers from the roster who we haven't 
160                 // exchanged ads with yet.
161                 if (this.model.getSettings().getMode() == Mode.get) {
162                     log.error("No associated ad for certificate!!");
163                 } else {
164                     log.debug("No kscope ad for cert in give mode, as expected");
165                 }
166                 this.proxyTracker.addJidProxy(jid);
167             }
168         }
169     }
