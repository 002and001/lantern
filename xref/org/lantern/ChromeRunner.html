44      private String determineExecutable() throws IOException {
45          final String path = determineExecutablePath();
46          final File file = new File(path);
47          if (!file.isFile()) {
48              throw new IOException("Could not find chrome at:" + path);
49          } else if (!file.canExecute()) {
50              throw new IOException("Chrome not executable at:" + path);
51          }
52          return path;
53      }

55      private String determineExecutablePath() {
56          if (SystemUtils.IS_OS_MAC_OSX) {
57              final File path1 = new File("install/osx/Lantern.app/Contents/MacOS/Lantern");
58              if (path1.isFile() && path1.canExecute()) return path1.getAbsolutePath();
59              final File path2 = new File("Lantern.app/Contents/MacOS/Lantern");
60              if (path2.isFile() && path2.canExecute()) return path2.getAbsolutePath();
61              throw new RuntimeException("Could not find LanternBrowser");
62              //chrome is broken on os x -- see #622
63              //return "/Applications/Google Chrome.app/Contents/MacOS/Google Chrome";
64          } else if (SystemUtils.IS_OS_LINUX) {
65              final String path1 = "/usr/bin/google-chrome";
66              final File opt1 = new File(path1);
67              if (opt1.isFile() && opt1.canExecute()) return path1;
68              final String path2 = "/usr/bin/chromium-browser";
69              final File opt2 = new File(path1);
70              if (opt2.isFile() && opt2.canExecute()) return path2;
71              throw new UnsupportedOperationException("Could not find chrome");
72          } else if (SystemUtils.IS_OS_WINDOWS) {
73              return findWindowsExe();
74          } else {
75              throw new UnsupportedOperationException(
76                      "Unsupported OS: "+SystemUtils.OS_NAME);
77          }
78      }

80      private String findWindowsExe() {//final String... opts) {
81          final Map<String, Integer> opts = new HashMap<String, Integer>();
82          opts.put("APPDATA", ShlObj.CSIDL_APPDATA);
83          opts.put("LOCALAPPDATA", ShlObj.CSIDL_LOCAL_APPDATA);
84          opts.put("PROGRAMFILES", ShlObj.CSIDL_PROGRAM_FILES);
85          opts.put("ProgramW6432", ShlObj.CSIDL_PROGRAM_FILESX86);
86          final String chromePath = "/Google/Chrome/Application/chrome.exe";
87          final Collection<String> paths = new HashSet<String>();
88          for (final Entry<String, Integer> entry : opts.entrySet()) {
89              final String base;
90              final String envBase = System.getenv(entry.getKey());
91              if (StringUtils.isBlank(envBase)) {
92                  base = Shell32Util.getFolderPath(entry.getValue().intValue());
93              } else {
94                  base = envBase;
95              }
96              if (StringUtils.isBlank(base)) {
97                  log.error("Could not resolve env variable: {}", base);
98                  continue;
99              }
100             final String path = base + chromePath;
101             paths.add(path);
102             final File candidate = new File(path);
103             if (candidate.isFile() && candidate.canExecute()) {
104                 log.debug("Running with path: {}", path);
105                 return path;
106             }
107         }
108         final String msg = 
109                 "Could not find Chrome on Windows!! Searched paths:\n"+paths;
110         log.error(msg);
111         throw new UnsupportedOperationException(msg);
112     }

118     public void open(final int port, String prefix) throws IOException {
119 
120         if (this.process != null) {
121             try {
122                 final int exitValue = this.process.exitValue();
123                 log.info("Got exit value from former process: ", exitValue);
124             } catch (final IllegalThreadStateException e) {
125                 // This indicates the existing process is still running.
126                 log.info("Ignoring open call since process is still running");
127                 return;
128             }
129         }
130         final String endpoint = StaticSettings.getLocalEndpoint(port, prefix);
131         log.info("Opening browser to: {}", endpoint);
132         final List<String> commands = new ArrayList<String>();
133         final String executable = determineExecutable();
134         commands.add(executable);
135         if (SystemUtils.IS_OS_MAC_OSX) {
136             commands.add(endpoint + "/index.html");
137         } else {
138             commands.add("--user-data-dir="
139                     + LanternClientConstants.CONFIG_DIR.getAbsolutePath());
140             commands.add("--window-size=" + screenWidth + "," + screenHeight);
141             commands.add("--window-position=" + location.x + "," + location.y);
142             commands.add("--disable-translate");
143             commands.add("--disable-sync");
144             commands.add("--no-default-browser-check");
145             commands.add("--disable-metrics");
146             commands.add("--disable-metrics-reporting");
147             commands.add("--temp-profile");
148             commands.add("--disable-plugins");
149             commands.add("--disable-java");
150             commands.add("--disable-extensions");
151             commands.add("--app=" + endpoint + "/index.html");
152         }
153 
154         final ProcessBuilder processBuilder = new ProcessBuilder(commands);
155 
156         // Note we don't call waitFor on the process to avoid blocking the
157         // calling thread and because we don't care too much about the return
158         // value.
159         this.process = processBuilder.start();
160         
161         new Analyzer(process.getInputStream());
162         new Analyzer(process.getErrorStream());
163     }

165     public void close() {
166         log.info("Closing Chrome browser...process is: {}", this.process);
167         if (this.process != null) {
168             log.info("Really closing Chrome browser...");
169             this.process.destroy();
170             this.process = null;
171         }
